/* Same as mpz_sizeinbase, meaning exact for power-of-2 bases, and either
   exact or 1 too big for other bases.  */

size_t
mpn_sizeinbase (mp_srcptr xp, mp_size_t xsize, int base)
{
  size_t  result;
  MPN_SIZEINBASE (result, xp, xsize, base);
  return result;
}
