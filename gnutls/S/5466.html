inline static int
_gnutls_pk_fixup(gnutls_pk_algorithm_t algo, gnutls_direction_t direction,
                 gnutls_pk_params_st * params)
{
        if (_gnutls_pk_ops.pk_fixup_private_params)
                return _gnutls_pk_ops.pk_fixup_private_params(algo,
                                                              direction,
                                                              params);
        return 0;
}
