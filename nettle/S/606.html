#define TMP_GMP_DECL(name, type) type *name;    \
  size_t tmp_##name##_size
#define TMP_GMP_ALLOC(name, size) do {                                  \
    tmp_##name##_size = (size);                                         \
    (name) = gmp_alloc(sizeof (*name) * (size));        \
  } while (0)
#define TMP_GMP_FREE(name) (gmp_free(name, tmp_##name##_size))
