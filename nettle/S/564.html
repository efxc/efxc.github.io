#include <stdio.h>
#include <stdlib.h>

#define GHASH_POLYNOMIAL 0xE1


/* When x is shifted out over the block edge, add multiples of the
   defining polynomial to eliminate each bit. */
static unsigned
reduce(unsigned x)
{
  unsigned p = GHASH_POLYNOMIAL << 1;
  unsigned y = 0;
  for (; x; x >>= 1, p <<= 1)
    if (x & 1)
      y ^= p;
  return y;
}

int
main(int argc, char **argv)
{
  unsigned i;
  printf("4-bit table:\n");
  
  for (i = 0; i<16; i++)
    {
      unsigned x;
      if (i && !(i%8))
        printf("\n");

      x = reduce(i << 4);
      printf("W(%02x,%02x),", x >> 8, x & 0xff);
    }
  printf("\n\n");
  printf("8-bit table:\n");
  for (i = 0; i<256; i++)
    {
      unsigned x;
      if (i && !(i%8))
        printf("\n");

      x = reduce(i);
      printf("W(%02x,%02x),", x >> 8, x & 0xff);
    }
  printf("\n");
  return EXIT_SUCCESS;
}
