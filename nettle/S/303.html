#include "base64.h"

void
base64url_encode_init(struct base64_encode_ctx *ctx)
{
  static const char base64url_encode_table[64] =
    "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    "abcdefghijklmnopqrstuvwxyz"
    "0123456789-_";

  ctx->word = ctx->bits = 0;
  ctx->alphabet = base64url_encode_table;
}
