/* Returns a suitable prime >= ESTIMATE.  */
static size_t
next_prime (size_t estimate)
{
  size_t i;

  for (i = 0; i < sizeof (primes) / sizeof (primes[0]); i++)
    if (primes[i] >= estimate)
      return primes[i];
  return SIZE_MAX; /* not a prime, but better than nothing */
}
